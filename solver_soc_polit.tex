
\subsection{Модель коррупции} \label{corruption_otvet_1}

\textbf{Вопрос \ref{corruption_vopros_1}.} Рост популярности правительства
%------------------------------------------------------------------
\begin{equation} \label{corruption_simple_rost}
  \dot{x} = \frac{\alpha x}{\beta + x} - \gamma y_0 z_0 x > 0
\end{equation}
%------------------------------------------------------------------
при $x \geqslant 0$, ввиду неравенства
$$
  \frac{\alpha}{\beta + x} < \frac{\alpha}{\beta + x_0}, 
$$
обеспечивается при выполнении соотношения
$$
\frac{\alpha}{\beta + x_0} - \gamma y_0 z_0 > 0.
$$
